  \hline 
  7 & 8 & 9 \\

  \end{verbatim}

    7 & 8 & 9 \\
    \hline
  \end{tabular}

	\begin{verbatim}
\usepackage{multirow}

  \multirow{2}{*}{Spanned rows} & 2 & 3\\

%%%'*' in witdh means using natural width    
  \cline{2-3}
                                & 5 & 6 \\
  \hline 
  7 & 8 & 9 \\
  \hline
\end{tabular}
	\end{verbatim}
}
